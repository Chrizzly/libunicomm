322 /*
323  * *getch*() -- a blocking single character input from stdin
324  *
325  * Returns a character, or -1 if an input error occurs.
326  *
327  *
328  * Conditionals allow compiling with or without echoing of
329  * the input characters, and with or without flushing pre-existing
330  * existing buffered input before blocking.
331  *
332  */
333 int getch(void)
334 {
335  char ch;
336  int error;
337  static struct termios Otty, Ntty;
338 
339  fflush(stdout);
340  tcgetattr(0, &Otty);
341  Ntty = Otty;
342 
343  Ntty.c_iflag = 0; /* input mode */
344  Ntty.c_oflag = 0; /* output mode */
345  Ntty.c_lflag &= ~ICANON; /* line settings */
346 
347 #if 1
348  /* disable echoing the char as it is typed */
349  Ntty.c_lflag &= ~ECHO; /* disable echo */
350 #else
351  /* enable echoing the char as it is typed */
352  Ntty.c_lflag |= ECHO; /* enable echo */
353 #endif
354 
355  Ntty.c_cc[VMIN] = CMIN; /* minimum time to wait */
356  Ntty.c_cc[VTIME] = CTIME; /* minimum chars to wait for */
357 
358 #if 1
359 /*
360  * use this to flush the input buffer before blocking for new input
361  */
362 #define FLAG TCSAFLUSH
363 #else
364 /*
365  * use this to return a char from the current input buffer, or block if
366  * no input is waiting.
367  */
368 #define FLAG TCSANOW
369 
370 #endif
371 
372  if (0 == (error = tcsetattr(0, FLAG, &Ntty))) {
373  error = read(0, &ch, 1 ); /* get char from stdin */
374  error += tcsetattr(0, FLAG, &Otty); /* restore old settings */
375  }
376 
377  return (error == 1 ? (int) ch : -1 );
378 
379 }

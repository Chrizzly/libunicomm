34 // Each protocol has messages, client and server. Let's define
35 // them. Messages are objects that hold data sent and received
36 // by client and server. In this example there is only the message.
37 // Each message class should define name() function returns the unique name.
38 // Here we use default defined one in the base class due to we only have
39 // one message.
40 
42 struct message : public unicomm::message_base
43 {
45 // interface
46 public:
47  explicit message(const std::string& s = ""): _data(s) { /*empty*/ }
48 
49  const std::string& data(void) const { return _data; }
50  void data(const std::string& s) { _data = s; }
51 
53 // private stuff
54 private:
56  std::string serialize(void) const { return data(); }
57 
59  void unserialize(const std::string &message) { data(message); }
60 
61 private:
62  std::string _data;
63 };

264 /*
265 * *getch*() -- a blocking single character input from stdin
266 *
267 * Returns a character, or -1 if an input error occurs.
268 *
269 *
270 * Conditionals allow compiling with or without echoing of
271 * the input characters, and with or without flushing pre-existing
272 * existing buffered input before blocking.
273 *
274 */
275 int getch(void)
276 {
277  char ch;
278  int error;
279  static struct termios Otty, Ntty;
280 
281  fflush(stdout);
282  tcgetattr(0, &Otty);
283  Ntty = Otty;
284 
285  Ntty.c_iflag = 0; /* input mode */
286  Ntty.c_oflag = 0; /* output mode */
287  Ntty.c_lflag &= ~ICANON; /* line settings */
288 
289 #if 1
290  /* disable echoing the char as it is typed */
291  Ntty.c_lflag &= ~ECHO; /* disable echo */
292 #else
293  /* enable echoing the char as it is typed */
294  Ntty.c_lflag |= ECHO; /* enable echo */
295 #endif
296 
297  Ntty.c_cc[VMIN] = CMIN; /* minimum time to wait */
298  Ntty.c_cc[VTIME] = CTIME; /* minimum chars to wait for */
299 
300 #if 1
301  /*
302  * use this to flush the input buffer before blocking for new input
303  */
304 #define FLAG TCSAFLUSH
305 #else
306  /*
307  * use this to return a char from the current input buffer, or block if
308  * no input is waiting.
309  */
310 #define FLAG TCSANOW
311 
312 #endif
313 
314  if (0 == (error = tcsetattr(0, FLAG, &Ntty))) {
315  error = read(0, &ch, 1 ); /* get char from stdin */
316  error += tcsetattr(0, FLAG, &Otty); /* restore old settings */
317  }
318 
319  return (error == 1 ? (int) ch : -1 );
320 
321 }
